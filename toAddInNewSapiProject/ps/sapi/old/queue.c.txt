/*---------------------------------------------------------------------------*/
/* queueInit                                                                 */
/*---------------------------------------------------------------------------*/

int32_t queueInit(queue_t * q)
{
	int32_t rv = -1;
	q->tail = 0;
	q->head = 0;
	q->task = INVALID_TASK;
	q->ev = os_event_init();
	if (q->ev != INVALID_EVENT) {
		rv = 0;
	}
	return rv;
}


/*---------------------------------------------------------------------------*/
/* queuePut                                                                  */
/*---------------------------------------------------------------------------*/

void queuePut(queue_t * q, queueItem_t d)
{
  // Mientras cola llena --> Bloquear la tarea (productora) que llamo a queuePut
	while(((q->head+1)%QUEUE_LEN) == q->tail)
	{
		// Si ya hay una tarea esperando
		if(q->task != INVALID_TASK)
		{
			/* error! esta queue soporta una tarea esperando a la vez */
			while (1) { }
		}

		// Obtengo la tarea que está usando la CPU (una productora).
		// La guardo en el parámetro que tiene la tarea que va a
		// esperar por lugar en la cola.
		q->task = os_get_running_task();

		// Pongo la tarea productora a la espera de un evento, este evento se
		// va a dar cuando haya lugar para guardar un dato en la cola.
		if (os_event_wait(q->ev) != 0) {
			/* error al esperar evento */
			while (1) { }
		}

		// Elimino la tarea a esperar de la cola.
		q->task = INVALID_TASK;
	}

  // Si salió del While entonces hay lugar para guardar un dato en la cola:

	// Pone el dato en la cola
	q->data[q->head] = d;

	// Incremento el head (cola circular)
	q->head = (q->head+1)%QUEUE_LEN;

	// Si la tarea a esperar de la cola es distinta de INVALID_TASK (entonces
	// hay alguna tarea esperando) le setea el evento desbloqueándola.
	// Esta tarea seria una consumidora que se quedo a la espera de que haya
	// algún dato para leer.
	if(q->task != INVALID_TASK)
	{
		if (os_event_set(q->ev) != 0) {
			/* error al setear evento */
			while (1) { }
		}
	}

}


/*---------------------------------------------------------------------------*/
/* queueGet                                                                  */
/*---------------------------------------------------------------------------*/

queueItem_t queueGet(queue_t * q)
{
  // Mientras cola vacia --> Bloquear la tarea (consumidora) que llamo a queueGet
	while(q->head == q->tail)
	{
		// Si ya hay una tarea esperando
		if(q->task != INVALID_TASK)
		{
			/* error! esta queue soporta una tarea esperando a la vez */
			while (1);
		}

		// Obtengo la tarea que está usando la CPU (una consumidora).
		// La guardo en el parámetro que tiene la tarea que va a
		// esperar por que haya algún dato en la cola.
		q->task = os_get_running_task();

		// Pongo la tarea consumidora a la espera del evento, este evento se
		// va a dar cuando haya algún dato para consumir en la cola.
		if (os_event_wait(q->ev) != 0) {
			/* error al esperar evento */
			while (1) { }
		}

		// Elimino la tarea a esperar de la cola.
		q->task = INVALID_TASK;
	}

  // Si salió del While entonces hay algún dato para consumir en la cola:

	// Leo de la cola
	queueItem_t d = q->data[q->tail];

	// Incremento el tail (cola circular)
	q->tail = (q->tail+1)%QUEUE_LEN;

	// Si la tarea a esperar de la cola es distinta de INVALID_TASK (entonces
	// hay alguna tarea esperando) le setea el evento desbloqueándola.
	// Esta tarea seria una productora que se quedó a la espera de lugar
	// para escribir.
	if(q->task != INVALID_TASK)
	{
		if (os_event_set(q->ev) != 0) {
			/* error al setear evento */
			while (1) { }
		}
	}
	return d;
}
